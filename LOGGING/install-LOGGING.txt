==Monitoring with Prometheus & Grafana==

curl -fsSL -o get_helm.sh https://raw.githubusercontent.com/helm/helm/master/scripts/get-helm-3
chmod 700 get_helm.sh
./get_helm.sh
helm version ( jika belum install helm )

helm repo add stable https://charts.helm.sh/stable
helm repo add prometheus-community https://prometheus-community.github.io/helm-charts
helm repo add grafana https://grafana.github.io/helm-charts
helm repo update
helm search repo prometheus-community
helm install stable prometheus-community/kube-prometheus-stack (--namespace untuk menginstall di ns kalian buat )
helm install loki loki/loki --namespace logging

curl https://raw.githubusercontent.com/grafana/loki/master/production/helm/promtail/values.yaml > values.yaml

edit di serviceName = loki

helm install promtail --namespace logging loki/promtail -f values.yaml
